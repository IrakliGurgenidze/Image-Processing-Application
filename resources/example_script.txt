load resources/mushrooms.jpg mushroom

#brighten mushroom by adding 10
brighten 10 mushroom mushroom-brighter

#flip mushroom vertically
vertical-flip mushroom mushroom-vertical

#flip the vertically flipped mushroom horizontally
horizontal-flip mushroom-vertical mushroom-vertical-horizontal

#create a greyscale using only the value component, as an image mushroom-greyscale
value-component mushroom mushroom-greyscale

#save mushroom-brighter
save resources/result_images/mushroom-brighter.jpg mushroom-brighter

#save mushroom-greyscale
save resources/result_images/mushroom-greyscale.ppm mushroom-greyscale

#overwrite mushroom with another file
load resources/mushrooms.jpg mushroom

#give the mushroom a red tint
rgb-split mushroom mushroom-red mushroom-green mushroom-blue

#brighten just the red image
brighten 50 mushroom-red mushroom-red-brighter

#combine them back, but by using the brightened red we get a red tint
rgb-combine mushroom-red-tint mushroom-red-brighter mushroom-green mushroom-blue

save resources/result_images/mushroom-red-tint.png mushroom-red-tint
